18  package net.sf.beezle.maven.plugins.application;
19  
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.HashMap;
23  import java.util.Iterator;
24  import java.util.List;
25  import java.util.Map;
26  
27  import net.sf.beezle.sushi.fs.Node;
28  import org.apache.maven.artifact.Artifact;
29  
30  public class Sources {
31      private Map<String, List<Artifact>> map;
32  
33      public Sources() {
34          this.map = new HashMap<String, List<Artifact>>();
35      }
36  
37      public void add(String path, Artifact source) {
38          List<Artifact> lst;
39  
40          lst = map.get(path);
41          if (lst == null) {
42              lst = new ArrayList<Artifact>();
43              map.put(path, lst);
44          }
45          lst.add(source);
46      }
47  
48      public void addAll(Node root, Artifact source) throws IOException {
49          for (Node node : root.find("**/*")) {
50              add(node.getRelative(root), source);
51          }
52      }
53  
54      public List<Artifact> get(String path) {
55          return map.get(path);
56      }
57  
58      public void retain(List<String> paths) {
59          Iterator<Map.Entry<String, List<Artifact>>> iter;
60          Map.Entry<String, List<Artifact>> entry;
61  
62          iter = map.entrySet().iterator();
63          while (iter.hasNext()) {
64              entry = iter.next();
65              if (!paths.contains(entry.getKey())) {
66                  iter.remove();
67              }
68          }
69      }
70  
71      public String toString() {
72          StringBuilder builder;
73          Iterator<Map.Entry<String, List<Artifact>>> iter;
74          Iterator<Map.Entry<String, List<Artifact>>> sub;
75          Map.Entry<String, List<Artifact>> entry;
76          List<Artifact> sources;
77  
78          builder = new StringBuilder();
79          while (true) {
80              iter = map.entrySet().iterator();
81              if (!iter.hasNext()) {
82                  break;
83              }
84              entry = iter.next();
85              sources = entry.getValue();
86              builder.append("  ").append(sources.toString()).append(":\n");
87              builder.append("    ").append(entry.getKey()).append('\n');
88              iter.remove();
89              sub = map.entrySet().iterator();
90              while (sub.hasNext()) {
91                  entry = sub.next();
92                  if (sources.equals(entry.getValue())) {
93                      builder.append("    ").append(entry.getKey()).append('\n');
94                      sub.remove();
95                  }
96              }
97          }
98          return builder.toString();
99      }
