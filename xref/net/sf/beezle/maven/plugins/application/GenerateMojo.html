215     public void doExecute() throws IOException, MojoExecutionException {
216         validate(name);
217         // do not wipe the directory because other plugins might already have copied stuff into it
218         dir.mkdirsOpt();
219         script();
220         jar();
221         getLog().info(">" + size(getFile().getFile()) + getFile());
222         verify();
223         projectHelper.attachArtifact(project, "sh", classifier, getFile().getFile());
224     }

226     private void verify() throws MojoExecutionException {
227         URL url;
228         URLClassLoader loader;
229         Class<?> clazz;
230 
231         try {
232             url = getFile().getFile().toURI().toURL();
233         } catch (MalformedURLException e) {
234             throw new IllegalStateException();
235         }
236         loader = new URLClassLoader(new URL[] { url });
237         try {
238             clazz = loader.loadClass(main);
239         } catch (ClassNotFoundException e) {
240             throw new MojoExecutionException("main class not found: " + main, e);
241         }
242         try {
243             clazz.getDeclaredMethod("main", String[].class);
244         } catch (NoSuchMethodException e) {
245             throw new MojoExecutionException("main class has no main(String[]) method: " + main, e);
246         }
247     }

249     private void script() throws IOException {
250         Node file;
251         List<String> lines;
252 
253         lines = new ArrayList<String>();
254         lines.addAll(Arrays.asList(
255                 "#!/bin/sh",
256                 // resolve symlinks
257                 "APP=\"$0\"",
258                 "while [ -h \"$APP\" ] ; do",
259                 "  ls=$(ls -ld \"$APP\")",
260                 "  link=$(expr \"$ls\" : '.*-> \\(.*\\)$')",
261                 "  if expr \"$link\" : '/.*' > /dev/null; then",
262                 "    APP=\"$link\"",
263                 "  else",
264                 "    APP=$(dirname \"$APP\")\"/$link\"",
265                 "  fi",
266                 "done",
267                 // will be overridden with the configured name
268                 "NAME=$(basename \"$APP\")",
269                 "APP=$(dirname \"$APP\")",
270                 "APP=$(cd \"$APP\" && pwd)",
271                 "APP=\"$APP/$NAME\"",
272 
273                 // make pom configuration available for "extensions:"
274                 "PATH=" + path,
275                 "JAVA=" + java,
276                 "OPTIONS=" + options,
277                 "MAIN=" + main,
278                 "NAME=" + name
279                 ));
280         lines.addAll(extensions);
281 
282         // Notes
283         // * $OPTIONS before getOptsVar to allow users to override built-in options
284         // * reference jar via $APP to have symbolic links eleminated
285         // * do not call with -jar to allow classpath modifications
286         lines.add("$JAVA $OPTIONS $" + getOptsVar() + " -cp \"$APP\" $MAIN \"$@\"");
287         // explicitly quit the script because I want to append to this file:
288         lines.add("exit $?");
289         file = getFile();
290         file.writeLines(lines);
291         file.setMode(0755);
292     }

304     private List<Artifact> getDependencies() {
305         List<Artifact> artifacts;
306         Artifact a;
307 
308         artifacts = new ArrayList<Artifact>();
309         for (Iterator<?> i = project.getArtifacts().iterator(); i.hasNext(); ) {
310             a = (Artifact) i.next();
311             if (scopes.contains(a.getScope())) {
312                 artifacts.add(a);
313             }
314         }
315         a = project.getArtifact();
316         if (a.getFile() == null) {
317             // happens if application:generate is called directly from the command line
318             a.setFile(new File(projectJar));
319         }
320         artifacts.add(a);
321         return artifacts;
322     }

324     public String getOptsVar() {
325         String suffix = "_OPTS";
326         StringBuilder builder;
327         char c;
328 
329         builder = new StringBuilder(name.length() + suffix.length());
330         for (int i = 0; i < name.length(); i++) {
331             c = name.charAt(i);
332             if (Character.isLetter(c) || (i > 0 && Character.isDigit(c))) {
333                 builder.append(Character.toUpperCase(c));
334             } else {
335                 builder.append('_');
336             }
337         }
338         builder.append(suffix);
339         return builder.toString();
340     }

342     public void jar() throws IOException, MojoExecutionException {
343         Archive archive;
344         OutputStream dest;
345         List<String> roots;
346 
347         archive = Archive.createJar(world);
348         addDependencies(archive);
349         if (!archive.data.join(main.replace('.', '/') + ".class").isFile()) {
350             throw new MojoExecutionException("main class not found: " + main);
351         }
352         mainAttributes(archive.manifest.getMainAttributes());
353         if (strip) {
354             roots = Separator.COMMA.split(extraRoots);
355             roots.add(main + ".main");
356             try {
357                 Stripper.run(archive, roots) /* TODO .warnings() */;
358             } catch (NotFoundException e) {
359                 throw new MojoExecutionException("class not found", e);
360             }
361         }
362         dest = getFile().createAppendStream();
363         archive.save(dest);
364         dest.close();
365     }

375     private void addDependencies(Archive archive) throws IOException, MojoExecutionException {
376         Document plexus;
377         Sources sources;
378         File file;
379         Archive add;
380         Node jar;
381         List<String> duplicatePaths;
382 
383         plexus = null;
384         sources = new Sources();
385         duplicatePaths = new ArrayList<String>();
386         for (Artifact artifact : getDependencies()) {
387             getLog().info("+" + size(artifact.getFile()) + gav(artifact));
388             file = artifact.getFile();
389             if (file == null) {
390                 throw new IllegalStateException("unresolved dependency: " + gav(artifact) + ".jar");
391             }
392             jar = world.file(file);
393             add = Archive.loadJar(jar);
394             sources.addAll(add.data, artifact);
395             removeFiles(add.data);
396             concat(add.data, archive.data);
397             copy(add.data, archive.data, duplicatePaths);
398             archive.mergeManifest(add.manifest);
399             plexus = plexusMerge(archive.data, plexus);
400         }
401         if (duplicatePaths.size() > 0) {
402             sources.retain(duplicatePaths);
403             throw new MojoExecutionException("duplicate files:\n" + sources.toString());
404         }
405         plexusSave(archive.data, plexus);
406     }

411     private void copy(Node srcdir, Node destdir, List<String> duplicates) throws IOException, MojoExecutionException {
412         List<Node> mayEqual;
413         List<Node> mayOverwrite;
414         Node destfile;
415         String relative;
416 
417         mayEqual = find(srcdir, equal);
418         mayOverwrite = find(srcdir, overwrite);
419         for (Node srcfile : srcdir.find("**/*")) {
420             relative = srcfile.getRelative(srcdir);
421             destfile = destdir.join(relative);
422             if (srcfile.isDirectory()) {
423                 destfile.mkdirsOpt();
424             } else {
425                 if (destfile.exists()) {
426                     if (srcfile.diff(destfile)) {
427                         if (mayOverwrite.contains(srcfile)) {
428                             getLog().debug("overwrite different " + relative);
429                             destfile.delete();
430                             srcfile.copyFile(destfile);
431                         } else {
432                             duplicates.add(relative);
433                         }
434                     } else {
435                         if (mayOverwrite.contains(srcfile)) {
436                             getLog().debug("overwrite equal " + relative);
437                         } else if (mayEqual.contains(srcfile)) {
438                             getLog().debug("equal " + relative);
439                         } else {
440                             duplicates.add(relative);
441                         }
442                     }
443                 } else {
444                     srcfile.copyFile(destfile);
445                 }
446             }
447         }
448     }

450     private static List<String> split(String str) throws MojoExecutionException {
451         List<String> result;
452 
453         result = Separator.COMMA.split(str);
454         for (String item : result) {
455             if (item.contains(" ")) {
456                 throw new MojoExecutionException(
457                         "Invalid space character in configuration value " + str + "\n" +
458                         "Use commas to separate multiple entries.");
459             }
460         }
461         return result;
462     }

504     private void concatOne(Node srcdir, Node destdir, String path) throws IOException {
505         Node src;
506         Node dest;
507         String lf;
508         StringBuilder builder;
509         int idx;
510 
511         src = srcdir.join(path);
512         dest = destdir.join(path);
513         if (!dest.exists()) {
514             dest.getParent().mkdirsOpt();
515             dest.writeString("");
516         }
517         lf = srcdir.getWorld().os.lineSeparator.getSeparator();
518         builder = new StringBuilder();
519         builder.append(src.readString());
520         idx = builder.lastIndexOf(lf);
521         if (idx + lf.length() != builder.length()) {
522             builder.append(lf);
523         }
524         builder.append(dest.readString());
525         dest.writeString(builder.toString());
526         src.delete();
527         getLog().debug("merged " + path + ":\n" + builder);
528     }

541     private Document plexusMerge(Node root, Document plexus) throws IOException {
542         Element componentSet;
543         Element components;
544         Node file;
545 
546         file = plexusFile(root);
547         if (!file.exists()) {
548             return plexus;
549         }
550         if (plexus == null) {
551             plexus = world.getXml().getBuilder().createDocument(ROOT);
552             components = Builder.element(plexus.getDocumentElement(), COMPONENTS);
553         } else {
554             components = (Element) plexus.getDocumentElement().getFirstChild();
555         }
556         try {
557             componentSet = file.readXml().getDocumentElement();
558         } catch (SAXException e) {
559             throw new IOException(file + ": " + e.getMessage());
560         }
561         if (!componentSet.getTagName().equals(ROOT)) {
562             throw new IOException(file + ": expected " + ROOT);
563         }
564         for (Element e : Dom.getAllChildElements(componentSet)) {
565             if (e.getTagName().equals(COMPONENTS)) {
566                 Builder.add(components, e.getChildNodes());
567             } else {
568                 throw new IOException("unknown element: " + e.getTagName());
569             }
570         }
571         file.delete();
572         return plexus;
573     }

595     public static String getUserEmail() {
596         String hostname;
597 
598         try {
599             hostname = InetAddress.getLocalHost().getHostName();
600         } catch (UnknownHostException e) {
601             // this happens if your hostname is not in /etc/hosts and not resolvable via dns.
