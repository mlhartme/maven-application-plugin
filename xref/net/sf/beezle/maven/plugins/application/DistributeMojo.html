18  package net.sf.beezle.maven.plugins.application;
19  
20  import java.io.IOException;
21  import java.net.URI;
22  import java.net.URISyntaxException;
23  import java.util.List;
24  
25  import net.sf.beezle.sushi.util.Separator;
26  import org.apache.maven.plugin.MojoExecutionException;
27  import net.sf.beezle.sushi.fs.World;
28  import net.sf.beezle.sushi.fs.Node;
29  import net.sf.beezle.sushi.util.Strings;
30  
31  /**
32   * Distributes application files.
33   *
34   * @goal distribute
35   */
36  public class DistributeMojo extends BaseMojo {
37      /**
38       * Comma separated list of targets. E.g. ssh://root@pumama.schlund.de/home/clustercontrol/launcher
39       *
40       * @parameter
41       * @required
42       */
43      private String distribute;
44  
45      /**
46       * Base name for backup file. If not specified, no backup will be created
47       *
48       * @parameter
49       */
50      private String backup;
51  
52      /**
53       * Pattern to search root directory of the respective distribute node for lock files.
54       * Distribution is aborted if the pattern matches one or more files.
55  
56       * @parameter
57       */
58      private String lockfile;
59  
60      public DistributeMojo() {
61          this(new World());
62      }
63  
64      public DistributeMojo(World world) {
65          super(world);
66      }
67  
68      public void execute() throws MojoExecutionException {
69          try {
70              doExecute();
71          } catch (IOException e) {
72              throw new MojoExecutionException("cannot distribute application: " + e.getMessage(), e);
73          }
74      }
75  
76      public void doExecute() throws IOException, MojoExecutionException {
77          URI uri;
78  
79          for (String machine : Separator.COMMA.split(distribute)) {
80              try {
81                  uri = new URI(machine);
82              } catch (URISyntaxException e) {
83                  throw new MojoExecutionException("invalid distribution target: " + machine.trim(), e);
84              }
85              deploy(world.node(uri));
86          }
87      }
88  
89      private void deploy(Node dest) throws IOException {
90          List<Node> lst;
91  
92          getLog().info("distributing to " + dest.getURI());
93          if (lockfile != null) {
94              lst = dest.getRootNode().find(lockfile);
95              if (lst.size() > 0) {
96                  throw new IOException("locked: " + lst);
97              }
98          }
99          if (backup != null && dest.join(name).exists()) {
100             move(dest.join(name), dest.join(backup));
101         }
102         copy(getFile(), dest.join(name));
103     }
104 
105     private void move(Node src, Node dest) throws IOException {
106         getLog().info("  move " + src.getURI() + " " + dest.getURI());
107         dest.deleteOpt();
108         src.move(dest);
109     }
110 
111     private void copy(Node src, Node dest) throws IOException {
112         getLog().info("  copy " + src.getURI() + " " + dest.getURI());
113         src.copyFile(dest);
114         dest.setMode(src.getMode());
115     }
116 }
